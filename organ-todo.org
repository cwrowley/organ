(defun my-package-make-tabulated-list-entries ()
  (let ((entries '()))
    (dolist (item my-package-data)
      (let ((name (propertize (car item) 'face 'my-package-highlight-face))
            (status (propertize (cadr item) 'face 'my-package-warning-face)))
        (push (list item (vector name status)) entries)))
    entries))

(defun add-new-gig ()
  "Interactively add a new gig by sending a POST request to the API."
  (interactive)
  (let ((date (read-string "Enter gig date (YYYY-MM-DD): "))
        (church (read-string "Enter church name: "))
        (pieces (read-string "Enter pieces (comma-separated): "))
    (request
     "http://localhost:1685/gigs"
     :type "POST"
     :headers '(("Content-Type" . "application/json"))
     :data (json-encode `(("date" . ,date)
                          ("church" . ,church)
                          ("pieces" . ,(split-string pieces ",\\s-*"))))
     :parser 'json-read
     :success (cl-function
               (lambda (&key data &allow-other-keys)
                 (message "Gig added successfully: %s" (alist-get 'id data))))
     :error (cl-function
             (lambda (&key error-thrown &allow-other-keys)
               (message "Error: %s" error-thrown)))))))

(defun fetch-pieces-and-format ()
  "Fetch the list of pieces from the API and return a list of cons cells.
Each cell has the form (\"composer - title\" . piece-id)."
  (interactive)
  (request
    "http://localhost:1685/pieces/"
   :type "GET"
   :headers '(("Accept" . "application/json"))
   :parser 'json-read
   :success (cl-function
             (lambda (&key data &allow-other-keys)
               (let ((formatted-pieces
                      (mapcar (lambda (piece)
                                (let ((title (alist-get 'title piece))
                                      (composer (alist-get 'composer piece))
                                      (id (alist-get 'id piece)))
                                  (cons (format "%s - %s" composer title) id)))
                              data)))
                 (message "Fetched pieces: %s" formatted-pieces)
                 formatted-pieces)))
   :error (cl-function
           (lambda (&key error-thrown &allow-other-keys)
             (message "Error fetching pieces: %s" error-thrown)))))

(defun select-piece ()
  "Prompt the user to select a piece from the cached list and return its ID.
Fetch pieces if the cache is empty."
  (interactive)
  (if my-pieces
      (let* ((completion-table (mapcar #'car my-pieces))
             (selected-piece (completing-read "Select a piece: " completion-table nil t)))
        (message "Selected piece ID: %s" (cdr (assoc selected-piece my-pieces)))
        (cdr (assoc selected-piece my-pieces)))
    (progn
      (message "Cache is empty, fetching pieces...")
      (refresh-pieces #'select-piece)
      (ignore))))

(defun add-new-piece ()

      :error (cl-function

(defun display-gigs-old ()
  "Fetch and display the list of gigs in a tabulated list mode buffer."
  (interactive)
  (organ--get-request "/gigs/"
   :success
   (cl-function
    (lambda (&key data &allow-other-keys)
      (setq gigs-data (append data nil))
      (let ((buffer (get-buffer-create "*Gigs*")))
        (with-current-buffer buffer
          (gigs-mode)
          (setq tabulated-list-entries (gigs-tabulated-list-entries))
          (tabulated-list-print t)
          (display-buffer buffer)))))))

  (request
