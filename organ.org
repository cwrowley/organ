#+begin_src python
import pytest
from fastapi.testclient import TestClient
from app.main import app

client = TestClient(app)

def test_read_gigs_empty():
    response = client.get("/gigs/")
    assert response.status_code == 200
    assert isinstance(response.json(), list)

def test_add_and_read_gig():
    # Add a church
    church_resp = client.post("/churches/", json={"name": "Test Church"})
    assert church_resp.status_code == 200
    church_id = church_resp.json()["id"]

    # Add a piece
    piece_resp = client.post("/pieces/", json={
        "title": "Test Piece",
        "composer": "Test Composer",
        "duration": 300
    })
    assert piece_resp.status_code == 200
    piece_id = piece_resp.json()["id"]

    # Add a gig
    gig_data = {
        "date": "2024-01-01",
        "church_id": church_id,
        "fee": 150.0,
        "pieces": [
            {"piece_id": piece_id, "role": "Prelude"}
        ]
    }
    gig_resp = client.post("/gigs/", json=gig_data)
    assert gig_resp.status_code == 200
    gig = gig_resp.json()
    assert gig["church"]["id"] == church_id
    assert gig["pieces"][0]["piece"]["id"] == piece_id
    assert gig["pieces"][0]["role"] == "Prelude"

    # Read the gig back
    gig_id = gig["id"]
    get_resp = client.get(f"/gigs/{gig_id}")
    assert get_resp.status_code == 200
    assert get_resp.json()["id"] == gig_id

def test_read_missing_gig():
    response = client.get("/gigs/999999")
    assert response.status_code == 404
    assert response.json()["detail"] == "Gig not found"

#+end_src
